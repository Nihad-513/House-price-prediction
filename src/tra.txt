def preprocess(df):
    df = df.copy()

    # Drop ID
    if "Id" in df.columns:
        df.drop("Id", axis=1, inplace=True)
    # Handle missing values
    df.fillna(df.median(numeric_only=True), inplace=True)
    df.fillna("None", inplace=True)

    # One-hot encoding
    df = pd.get_dummies(df)

    return df
